   <script type="text/babel">
       const { useState, useEffect, useMemo } = React;

       // --- 單字資料庫 (可擴充) ---
       const initialVocabData = [
           { id: 1, unit: 13, word: "significant", pos: "adj.", detail: "顯著的；重要的", derivatives: "significantly (adv.)", phrases: "a significant increase", example: "There has been a significant change in the climate.", type: "主要" },
           { id: 2, unit: 13, word: "calculate", pos: "v.", detail: "計算；估計", derivatives: "calculation (n.)", phrases: "calculate the cost", example: "We need to calculate the total area.", type: "主要" },
           { id: 3, unit: 14, word: "emergency", pos: "n.", detail: "緊急狀況", derivatives: "emergent (adj.)", phrases: "emergency exit", example: "In case of emergency, call 911.", type: "主要" },
           { id: 4, unit: 14, word: "urgent", pos: "adj.", detail: "緊急的；急迫的", derivatives: "urgently (adv.)", phrases: "urgent matter", example: "The problem requires urgent attention.", type: "主要" },
           { id: 5, unit: 15, word: "adventure", pos: "n.", detail: "冒險", derivatives: "adventurous (adj.)", phrases: "go on an adventure", example: "They set out on an adventure across the ocean.", type: "主要" },
           { id: 6, unit: 16, word: "benefit", pos: "n./v.", detail: "好處；獲益", derivatives: "beneficial (adj.)", phrases: "benefit from", example: "The new rule will benefit everyone.", type: "主要" },
           { id: 7, unit: 17, word: "flexible", pos: "adj.", detail: "有彈性的；靈活的", derivatives: "flexibility (n.)", phrases: "flexible schedule", example: "My working hours are very flexible.", type: "主要" },
           { id: 8, unit: 18, word: "guarantee", pos: "v./n.", detail: "保證", derivatives: "guarantor (n.)", phrases: "under guarantee", example: "We guarantee that the product is fresh.", type: "補充" },
           // 這裡可以繼續添加 Unit 13-18 的單字...
       ];

       const App = () => {
           const [view, setView] = useState('dashboard'); // dashboard, quiz-setup, quiz, results, dictionary, stars
           const [vocab, setVocab] = useState(initialVocabData);
           const [starredIds, setStarredIds] = useState(JSON.parse(localStorage.getItem('starredVocab') || '[]'));
           const [wrongIds, setWrongIds] = useState(JSON.parse(localStorage.getItem('wrongVocab') || '[]'));
           const [streak, setStreak] = useState(Number(localStorage.getItem('streak') || 0));
           const [lastLogin, setLastLogin] = useState(localStorage.getItem('lastLogin'));

           // Quiz States
           const [quizConfig, setQuizConfig] = useState({
               units: [13, 14, 15, 16, 17, 18],
               count: 10,
               shuffle: true,
               types: ["主要", "補充", "衍生"],
               mode: 'mixed'
           });
           const [currentQuiz, setCurrentQuiz] = useState([]);
           const [quizIndex, setQuizIndex] = useState(0);
           const [quizResults, setQuizResults] = useState([]);
           const [startTime, setStartTime] = useState(null);

           useEffect(() => {
               const today = new Date().toLocaleDateString();
               if (lastLogin !== today) {
                   if (lastLogin === new Date(Date.now() - 86400000).toLocaleDateString()) {
                       setStreak(prev => prev + 1);
                   } else {
                       setStreak(1);
                   }
                   setLastLogin(today);
                   localStorage.setItem('streak', streak + 1);
                   localStorage.setItem('lastLogin', today);
               }
           }, []);

           useEffect(() => {
               localStorage.setItem('starredVocab', JSON.stringify(starredIds));
           }, [starredIds]);

           const toggleStar = (id) => {
               setStarredIds(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]);
           };

           // --- 測驗引擎邏輯 ---
           const startQuiz = () => {
               let pool = vocab.filter(v => quizConfig.units.includes(v.unit) && quizConfig.types.includes(v.type));
               if (quizConfig.shuffle) pool = [...pool].sort(() => Math.random() - 0.5);
               
               const selected = pool.slice(0, quizConfig.count).map(item => {
                   const typeRoll = Math.random();
                   let qType = 'en_to_ch';
                   if (typeRoll > 0.7) qType = 'fill_choice';
                   else if (typeRoll > 0.5) qType = 'ch_to_en';
                   else if (typeRoll > 0.3) qType = 'fill_blank';

                   // 生成干擾項
                   const distractors = vocab
                       .filter(v => v.id !== item.id && v.pos === item.pos)
                       .sort(() => Math.random() - 0.5)
                       .slice(0, 3)
                       .map(v => v.word);

                   return { ...item, qType, options: [...distractors, item.word].sort(() => Math.random() - 0.5) };
               });

               setCurrentQuiz(selected);
               setQuizIndex(0);
               setQuizResults([]);
               setStartTime(Date.now());
               setView('quiz');
           };

           const submitAnswer = (answer) => {
               const isCorrect = answer.toLowerCase().trim() === currentQuiz[quizIndex].word.toLowerCase();
               const newResult = {
                   item: currentQuiz[quizIndex],
                   userAnswer: answer,
                   isCorrect
               };
               setQuizResults([...quizResults, newResult]);
               
               if (!isCorrect) {
                   setWrongIds(prev => [...new Set([...prev, currentQuiz[quizIndex].id])]);
                   localStorage.setItem('wrongVocab', JSON.stringify([...new Set([...wrongIds, currentQuiz[quizIndex].id])]));
               }

               if (quizIndex < currentQuiz.length - 1) {
                   setQuizIndex(quizIndex + 1);
               } else {
                   setView('results');
               }
           };

           // --- UI 組件 ---

           const Nav = () => (
               <nav className="flex justify-between items-center p-4 bg-white shadow-sm mb-6 sticky top-0 z-50">
                   <div className="flex items-center gap-2 cursor-pointer" onClick={() => setView('dashboard')}>
                       <div className="bg-indigo-600 text-white p-2 rounded-lg"><i className="fas fa-book-open"></i></div>
                       <h1 className="text-xl font-bold text-slate-800">極致單字王</h1>
                   </div>
                   <div className="flex gap-4 items-center">
                       <div className="text-orange-500 font-bold flex items-center gap-1">
                           <i className="fas fa-fire streak-fire"></i> {streak} 天
                       </div>
                       <button onClick={() => setView('dictionary')} className="text-slate-600 hover:text-indigo-600"><i className="fas fa-search"></i></button>
                       <button onClick={() => setView('stars')} className="text-slate-600 hover:text-yellow-500"><i className="fas fa-star"></i></button>
                   </div>
               </nav>
           );

           const Dashboard = () => (
               <div className="max-w-4xl mx-auto p-4 animate-fadeIn">
                   <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-8">
                       <div className="glass p-8 rounded-3xl border border-slate-100 card-hover bg-gradient-to-br from-indigo-50 to-white">
                           <h2 className="text-2xl font-bold mb-2">準備好挑戰了嗎？</h2>
                           <p className="text-slate-500 mb-6">今日目標：複習 20 個單字</p>
                           <button onClick={() => setView('quiz-setup')} className="bg-indigo-600 text-white px-8 py-3 rounded-2xl font-bold hover:bg-indigo-700 transition shadow-lg shadow-indigo-200">
                               開始測驗 <i className="fas fa-arrow-right ml-2"></i>
                           </button>
                       </div>
                       <div className="grid grid-cols-2 gap-4">
                           <div className="glass p-4 rounded-2xl border border-slate-100 flex flex-col items-center justify-center text-center">
                               <span className="text-slate-400 text-sm">已掌握單字</span>
                               <span className="text-2xl font-bold text-indigo-600">{vocab.length - wrongIds.length}</span>
                           </div>
                           <div className="glass p-4 rounded-2xl border border-slate-100 flex flex-col items-center justify-center text-center">
                               <span className="text-slate-400 text-sm">收藏單字</span>
                               <span className="text-2xl font-bold text-yellow-500">{starredIds.length}</span>
                           </div>
                           <div className="glass p-4 rounded-2xl border border-slate-100 flex flex-col items-center justify-center text-center">
                               <span className="text-slate-400 text-sm">連續登入</span>
                               <span className="text-2xl font-bold text-orange-500">{streak}</span>
                           </div>
                           <div className="glass p-4 rounded-2xl border border-slate-100 flex flex-col items-center justify-center text-center">
                               <span className="text-slate-400 text-sm">需強化</span>
                               <span className="text-2xl font-bold text-red-500">{wrongIds.length}</span>
                           </div>
                       </div>
                   </div>
                   
                   <h3 className="font-bold text-slate-800 mb-4 text-lg">快速特訓</h3>
                   <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
                       <button onClick={() => {setQuizConfig({...quizConfig, types: ["收藏"]}); startQuiz();}} className="p-4 bg-white rounded-2xl shadow-sm border border-slate-50 hover:bg-yellow-50 flex items-center gap-3 transition">
                           <i className="fas fa-star text-yellow-500"></i> 收藏夾複習
                       </button>
                       <button onClick={() => {setQuizConfig({...quizConfig, types: ["錯題"]}); startQuiz();}} className="p-4 bg-white rounded-2xl shadow-sm border border-slate-50 hover:bg-red-50 flex items-center gap-3 transition">
                           <i className="fas fa-times-circle text-red-500"></i> 錯題重測
                       </button>
                       <button className="p-4 bg-white rounded-2xl shadow-sm border border-slate-50 hover:bg-indigo-50 flex items-center gap-3 transition">
                           <i className="fas fa-bolt text-indigo-500"></i> 每日挑戰
                       </button>
                   </div>
               </div>
           );

           const QuizSetup = () => (
               <div className="max-w-2xl mx-auto p-6 glass rounded-3xl shadow-xl animate-fadeIn">
                   <h2 className="text-2xl font-bold mb-6 text-center">測驗設定</h2>
                   <div className="space-y-6">
                       <div>
                           <label className="block text-sm font-medium text-slate-700 mb-2">選擇範圍 (Units)</label>
                           <div className="flex flex-wrap gap-2">
                               {[13, 14, 15, 16, 17, 18].map(u => (
                                   <button
                                       key={u}
                                       onClick={() => setQuizConfig(prev => ({...prev, units: prev.units.includes(u) ? prev.units.filter(x => x !== u) : [...prev.units, u]}))}
                                       className={`px-4 py-2 rounded-xl border transition ${quizConfig.units.includes(u) ? 'bg-indigo-600 text-white border-indigo-600' : 'bg-white text-slate-600 border-slate-200'}`}
                                   >
                                       Unit {u}
                                   </button>
                               ))}
                           </div>
                       </div>
                       <div className="grid grid-cols-2 gap-4">
                           <div>
                               <label className="block text-sm font-medium text-slate-700 mb-2">題數</label>
                               <input type="number" value={quizConfig.count} onChange={e => setQuizConfig({...quizConfig, count: parseInt(e.target.value)})} className="w-full p-3 rounded-xl border border-slate-200" />
                           </div>
                           <div>
                               <label className="block text-sm font-medium text-slate-700 mb-2">順序</label>
                               <select className="w-full p-3 rounded-xl border border-slate-200" onChange={e => setQuizConfig({...quizConfig, shuffle: e.target.value === 'true'})}>
                                   <option value="true">隨機打亂</option>
                                   <option value="false">依序排列</option>
                               </select>
                           </div>
                       </div>
                       <button onClick={startQuiz} className="w-full py-4 bg-indigo-600 text-white rounded-2xl font-bold text-lg hover:bg-indigo-700 transition">開始測驗</button>
                   </div>
               </div>
           );

           const QuizScreen = () => {
               const q = currentQuiz[quizIndex];
               const [input, setInput] = useState('');
               
               const handleNext = () => {
                   submitAnswer(input);
                   setInput('');
               };

               return (
                   <div className="max-w-xl mx-auto p-6 animate-fadeIn">
                       <div className="flex justify-between items-center mb-8">
                           <span className="text-slate-400 font-medium">Question {quizIndex + 1} of {currentQuiz.length}</span>
                           <div className="w-32 h-2 bg-slate-200 rounded-full overflow-hidden">
                               <div className="h-full bg-indigo-600 transition-all duration-300" style={{width: `${((quizIndex + 1) / currentQuiz.length) * 100}%`}}></div>
                           </div>
                       </div>

                       <div className="glass p-8 rounded-3xl shadow-lg text-center mb-6 min-h-[300px] flex flex-col justify-center">
                           <div className="mb-2 text-indigo-600 font-bold uppercase tracking-wider text-sm">{q.pos}</div>
                           
                           {q.qType === 'en_to_ch' && <h2 className="text-4xl font-bold text-slate-800 mb-8">{q.word}</h2>}
                           {q.qType === 'ch_to_en' && <h2 className="text-4xl font-bold text-slate-800 mb-8">{q.detail}</h2>}
                           {(q.qType === 'fill_blank' || q.qType === 'fill_choice') && (
                               <div className="text-2xl text-slate-700 leading-relaxed mb-8">
                                   {q.example.split(new RegExp(`(${q.word})`, 'gi')).map((part, i) =>
                                       part.toLowerCase() === q.word.toLowerCase() ? <span key={i} className="border-b-2 border-indigo-400 px-2 text-indigo-600 font-bold">_____</span> : part
                                   )}
                                   <p className="text-base text-slate-400 mt-4 italic font-normal">({q.detail})</p>
                               </div>
                           )}

                           {q.qType === 'fill_choice' ? (
                               <div className="grid grid-cols-2 gap-3">
                                   {q.options.map(opt => (
                                       <button key={opt} onClick={() => submitAnswer(opt)} className="p-4 border-2 border-slate-100 rounded-2xl hover:border-indigo-400 hover:bg-indigo-50 transition font-medium text-slate-700">{opt}</button>
                                   ))}
                               </div>
                           ) : (
                               <div className="space-y-4">
                                   <input
                                       autoFocus
                                       type="text"
                                       value={input}
                                       onChange={e => setInput(e.target.value)}
                                       onKeyDown={e => e.key === 'Enter' && handleNext()}
                                       placeholder={q.qType === 'en_to_ch' ? "輸入中文意思" : "Type the English word..."}
                                       className="w-full p-4 text-center text-xl rounded-2xl border-2 border-slate-200 focus:border-indigo-600 outline-none transition"
                                   />
                                   <div className="flex gap-2">
                                       <button onClick={() => setInput(q.word[0])} className="flex-1 py-2 text-sm text-indigo-600 bg-indigo-50 rounded-xl">提示首字母</button>
                                       <button onClick={handleNext} className="flex-[2] py-4 bg-indigo-600 text-white rounded-2xl font-bold">送出</button>
                                   </div>
                               </div>
                           )}
                       </div>
                       <button onClick={() => toggleStar(q.id)} className={`w-full py-2 flex items-center justify-center gap-2 ${starredIds.includes(q.id) ? 'text-yellow-500' : 'text-slate-400'}`}>
                           <i className={`${starredIds.includes(q.id) ? 'fas' : 'far'} fa-star`}></i> 標記此單字
                       </button>
                   </div>
               );
           };

           const ResultsScreen = () => {
               const correctCount = quizResults.filter(r => r.isCorrect).length;
               const score = Math.round((correctCount / quizResults.length) * 100);
               const timeTaken = Math.round((Date.now() - startTime) / 1000);

               return (
                   <div className="max-w-2xl mx-auto p-4 animate-fadeIn">
                       <div className="text-center mb-8">
                           <h2 className="text-3xl font-bold text-slate-800">測驗完成！</h2>
                           <div className="flex justify-center gap-8 mt-6">
                               <div className="text-center">
                                   <div className="text-4xl font-black text-indigo-600">{score}</div>
                                   <div className="text-slate-400 text-sm">分數</div>
                               </div>
                               <div className="text-center">
                                   <div className="text-4xl font-black text-slate-800">{timeTaken}s</div>
                                   <div className="text-slate-400 text-sm">用時</div>
                               </div>
                               <div className="text-center">
                                   <div className="text-4xl font-black text-green-500">{correctCount}/{quizResults.length}</div>
                                   <div className="text-slate-400 text-sm">正確率</div>
                               </div>
                           </div>
                       </div>

                       <div className="space-y-3 mb-8">
                           {quizResults.map((res, i) => (
                               <div key={i} className={`p-4 rounded-2xl border flex items-center justify-between ${res.isCorrect ? 'bg-green-50 border-green-100' : 'bg-red-50 border-red-100'}`}>
                                   <div>
                                       <div className="font-bold text-slate-800">{res.item.word}</div>
                                       <div className="text-sm text-slate-500">{res.item.detail}</div>
                                   </div>
                                   <div className="text-right">
                                       <div className={`text-sm ${res.isCorrect ? 'text-green-600' : 'text-red-600'}`}>
                                           {res.isCorrect ? '正確' : `錯誤: ${res.userAnswer || '(未填)'}`}
                                       </div>
                                       {!res.isCorrect && <div className="text-xs font-bold text-slate-400">正確答案：{res.item.word}</div>}
                                   </div>
                               </div>
                           ))}
                       </div>

                       <button onClick={() => setView('dashboard')} className="w-full py-4 bg-slate-800 text-white rounded-2xl font-bold hover:bg-slate-900 transition">返回主頁</button>
                   </div>
               );
           };

           const Dictionary = () => {
               const [searchTerm, setSearchTerm] = useState('');
               const filtered = vocab.filter(v => v.word.includes(searchTerm) || v.detail.includes(searchTerm));

               return (
                   <div className="max-w-4xl mx-auto p-4 animate-fadeIn">
                       <div className="sticky top-20 bg-f8fafc py-4 z-10">
                           <input
                               type="text"
                               placeholder="搜尋單字或中文..."
                               className="w-full p-4 rounded-2xl border-2 border-slate-100 shadow-sm focus:border-indigo-400 outline-none"
                               onChange={e => setSearchTerm(e.target.value)}
                           />
                       </div>
                       <div className="space-y-4 mt-4">
                           {filtered.map(v => (
                               <div key={v.id} className="glass p-6 rounded-3xl border border-slate-100 flex flex-col md:flex-row md:items-center justify-between gap-4 card-hover">
                                   <div className="flex-1">
                                       <div className="flex items-center gap-3 mb-1">
                                           <h3 className="text-2xl font-bold text-indigo-600">{v.word}</h3>
                                           <span className="text-xs px-2 py-1 bg-slate-100 text-slate-500 rounded-md">{v.pos}</span>
                                           <span className="text-xs px-2 py-1 bg-indigo-50 text-indigo-500 rounded-md">Unit {v.unit}</span>
                                       </div>
                                       <p className="text-lg text-slate-700 font-medium mb-2">{v.detail}</p>
                                       <div className="grid grid-cols-1 md:grid-cols-2 gap-2 text-sm">
                                           <div className="text-slate-500 italic"><span className="font-bold text-slate-400">例句:</span> {v.example}</div>
                                           <div className="text-slate-500"><span className="font-bold text-slate-400">搭配:</span> {v.phrases}</div>
                                       </div>
                                   </div>
                                   <button onClick={() => toggleStar(v.id)} className={`text-2xl ${starredIds.includes(v.id) ? 'text-yellow-500' : 'text-slate-200'}`}>
                                       <i className="fas fa-star"></i>
                                   </button>
                               </div>
                           ))}
                       </div>
                   </div>
               );
           };

           return (
               <div className="pb-10">
                   <Nav />
                   {view === 'dashboard' && <Dashboard />}
                   {view === 'quiz-setup' && <QuizSetup />}
                   {view === 'quiz' && <QuizScreen />}
                   {view === 'results' && <ResultsScreen />}
                   {view === 'dictionary' && <Dictionary />}
                   {view === 'stars' && <Dictionary />}
               </div>
           );
       };

       const root = ReactDOM.createRoot(document.getElementById('root'));
       root.render(<App />);
   </script>
